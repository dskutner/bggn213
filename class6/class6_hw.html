protein <- function(code, chain = "A", elety="CA", typ="l") {  # "code" is the main argument of the function, requiring the PDB filename or the four letter PDB identifier. "chain", "elety" and "typ" have defaults, but can be changed
  pdb <- read.pdb(code)                                    # reads the pdb code for the desired protein
  chain <- trim.pdb(pdb, chain=chain, elety=elety)         # loads the chain information
  plotb3(chain$atom$b, sse=chain, typ=typ, ylab="Bfactor", main = paste("PDB:", code)) # returns the plot of B-factor values along with secondary structure from PDB. The type of plot is changeable (e.g. line, points, etc.).
  }
